<script>
function refreshCaptcha() {
    fetch('{{ route('captcha.refresh') }}')
        .then(response => response.json())
        .then(data => {
            document.getElementById('captcha-display').textContent = data.captcha;
        })
        .catch(error => {
            console.error('Error refreshing captcha:', error);
        });
}
</script>
